Here are the steps to add the corrupted replica to the existing Always-ON availability group(s).

I have two availability groups: AG (AdventureWorks2019) and AG2 (AdventureWorksDW2019).

```sql
:CONNECT SQLSERVERVM02
Alter availability group [AG] Failover
GO
Alter availability group [AG2] Failover
GO
ALTER AVAILABILITY GROUP [AG] REMOVE REPLICA ON N'SQLSERVERVM01';
GO
ALTER AVAILABILITY GROUP [AG2] REMOVE REPLICA ON N'SQLSERVERVM01';
GO
:Connect SQLSERVERVM02
use [master]
GO
GRANT CONNECT ON ENDPOINT::[Hadr_endpoint] TO [TLMS\mssql]
GO
:Connect SQLSERVERVM03
use [master]
GO
GRANT CONNECT ON ENDPOINT::[Hadr_endpoint] TO [TLMS\mssql]
GO
:Connect SQLSERVERVM01
USE [master]
GO
CREATE ENDPOINT [Hadr_endpoint] 
	AS TCP (LISTENER_PORT = 5022)
	FOR DATA_MIRRORING (ROLE = ALL, ENCRYPTION = REQUIRED ALGORITHM AES)
GO
IF (SELECT state FROM sys.endpoints WHERE name = N'Hadr_endpoint') <> 0
BEGIN
	ALTER ENDPOINT [Hadr_endpoint] STATE = STARTED
END
GO
use [master]
GO
GRANT CONNECT ON ENDPOINT::[Hadr_endpoint] TO [TLMS\mssql]
GO
:Connect SQLSERVERVM01
IF EXISTS(SELECT * FROM sys.server_event_sessions WHERE name='AlwaysOn_health')
BEGIN
  ALTER EVENT SESSION [AlwaysOn_health] ON SERVER WITH (STARTUP_STATE=ON);
END
IF NOT EXISTS(SELECT * FROM sys.dm_xe_sessions WHERE name='AlwaysOn_health')
BEGIN
  ALTER EVENT SESSION [AlwaysOn_health] ON SERVER STATE=START;
END
GO
:Connect SQLSERVERVM02
USE [master]
GO
ALTER AVAILABILITY GROUP [AG]
ADD REPLICA ON N'SQLSERVERVM01' 
WITH (
ENDPOINT_URL = N'TCP://SQLServerVM01.TLMS.com:5022', 
FAILOVER_MODE = MANUAL, 
AVAILABILITY_MODE = SYNCHRONOUS_COMMIT, 
BACKUP_PRIORITY = 50, 
SECONDARY_ROLE(ALLOW_CONNECTIONS = ALL));

GO

ALTER AVAILABILITY GROUP [AG2]
ADD REPLICA ON N'SQLSERVERVM01' 
WITH (
ENDPOINT_URL = N'TCP://SQLServerVM01.TLMS.com:5022', 
FAILOVER_MODE = MANUAL, 
AVAILABILITY_MODE = SYNCHRONOUS_COMMIT, 
BACKUP_PRIORITY = 50, 
SECONDARY_ROLE(ALLOW_CONNECTIONS = ALL));

GO
:Connect SQLSERVERVM01
ALTER AVAILABILITY GROUP [AG] JOIN;
GO
ALTER AVAILABILITY GROUP [AG2] JOIN;
GO

:CONNECT SQLSERVERVM02
BACKUP DATABASE [AdventureWorks2019] 
TO DISK = N'\\192.168.100.101\share\AdventureWorks2019_master_corrupt.bak' WITH NOFORMAT, INIT,  
NAME = N'AdventureWorks2019-Full Database Backup', SKIP, NOREWIND, NOUNLOAD, COMPRESSION, STATS = 10
GO
BACKUP log [AdventureWorks2019] 
TO DISK = N'\\192.168.100.101\share\AdventureWorks2019_master_corrupt.bak' WITH NOFORMAT, NOINIT,  
NAME = N'AdventureWorks2019-Full Database Backup', SKIP, NOREWIND, NOUNLOAD, COMPRESSION, STATS = 10
GO
:CONNECT SQLSERVERVM01
RESTORE DATABASE [AdventureWorks2019] 
FROM DISK = N'\\192.168.100.101\share\AdventureWorks2019_master_corrupt.bak' 
WITH FILE = 1, REPLACE, NORECOVERY,  NOUNLOAD,  STATS = 5
GO
RESTORE LOG [AdventureWorks2019] 
FROM DISK = N'\\192.168.100.101\share\AdventureWorks2019_master_corrupt.bak' 
WITH FILE = 2, NOUNLOAD, NORECOVERY, STATS = 5
GO
ALTER DATABASE [AdventureWorks2019] SET HADR AVAILABILITY GROUP = [AG]
go

:CONNECT SQLSERVERVM02
BACKUP DATABASE [AdventureWorksDW2019] 
TO DISK = N'\\192.168.100.101\share\AdventureWorksDW2019_master_corrupt.bak' WITH NOFORMAT, INIT,  
NAME = N'AdventureWorksDW2019-Full Database Backup', SKIP, NOREWIND, NOUNLOAD, COMPRESSION, STATS = 10
GO
BACKUP log [AdventureWorksDW2019] 
TO DISK = N'\\192.168.100.101\share\AdventureWorksDW2019_master_corrupt.bak' WITH NOFORMAT, NOINIT,  
NAME = N'AdventureWorksDW2019-Full Database Backup', SKIP, NOREWIND, NOUNLOAD, COMPRESSION, STATS = 10
GO
:CONNECT SQLSERVERVM01
RESTORE DATABASE [AdventureWorksDW2019] 
FROM DISK = N'\\192.168.100.101\share\AdventureWorksDW2019_master_corrupt.bak' 
WITH FILE = 1, REPLACE, NORECOVERY,  NOUNLOAD,  STATS = 5
GO
RESTORE LOG [AdventureWorksDW2019] 
FROM DISK = N'\\192.168.100.101\share\AdventureWorksDW2019_master_corrupt.bak' 
WITH FILE = 2, NOUNLOAD, NORECOVERY, STATS = 5
GO
ALTER DATABASE [AdventureWorksDW2019] SET HADR AVAILABILITY GROUP = [AG2]
go

```
